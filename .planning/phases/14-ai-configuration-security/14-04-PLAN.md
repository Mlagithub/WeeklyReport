    ```python
    import requests
    from requests.exceptions import RequestException, Timeout, ConnectionError

    # ... existing functions ...

    def test_ai_connection(api_url: str, api_key: str, timeout: int = 30):
        """Test connection to AI API endpoint.

        Per CONFIG-02: Verify AI service availability.
        Per RESEARCH.md: Use GET /models endpoint (lightweight, no token consumption).

        Args:
            api_url: Base API URL (e.g., 'https://api.openai.com/v1')
            api_key: Plain text API key for authentication
            timeout: Request timeout in seconds (default 30)

        Returns:
            tuple: (success: bool, message: str)
            - (True, "连接成功") on successful connection
            - (False, "连接失败：{具体错误信息}") on failure with Chinese error
        """
        if not api_url or not api_key:
            return (False, "连接失败：API URL或API Key未配置")

        # Ensure URL ends without trailing slash for consistent path joining
        base_url = api_url.rstrip('/')
        models_url = f"{base_url}/models"

        headers = {
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json"
        }

        try:
            response = requests.get(
                models_url,
                headers=headers,
                timeout=timeout
            )

            if response.status_code == 200:
                return (True, "连接成功")
            elif response.status_code == 401:
                return (False, "连接失败：API Key无效")
            elif response.status_code == 404:
                return (False, "连接失败：API URL不正确")
            elif response.status_code == 429:
                return (False, "连接失败：请求过于频繁，请稍后再试")
            else:
                return (False, f"连接失败：服务器返回错误 {response.status_code}")

        except ConnectionError:
            return (False, "连接失败：网络连接失败，请检查API URL")
        except Timeout:
            return (False, "连接失败：请求超时，请检查网络或API服务状态")
        except RequestException as e:
            return (False, f"连接失败：{str(e)}")
        except Exception as e:
            return (False, f"连接失败：未知错误")
    ```

    Note: requests library is standard Python HTTP client, already available via weasyprint dependencies.
    Error messages in Chinese per REQUIREMENTS.md API-02 patterns.

    Complete route structure:
    ```python
    @app.route("/ai-config", methods=["GET", "POST"])
    @login_required
    @with_db_transaction
    def ai_config():
        """AI configuration route - admin only."""
        if not current_user.is_admin:
            flash("需要管理员权限", "warning")
            return redirect(url_for("home"))

        form = AIConfigForm()
        config = AIConfig.get_config()

        if form.validate_on_submit():
            if form.test_submit.data:  # Test connection button
                success, message = test_ai_connection(
                    form.api_url.data,
                    form.api_key.data
                )
                if success:
                    flash(message, "success")
                else:
                    flash(message, "danger")
                config = AIConfig.get_config()
                return render_template("config.html", ai_form=form, ai_config=config, form=form)

            if form.submit.data:  # Save button
                encrypted_key = encrypt_api_key(form.api_key.data)
                if config:
                    config.api_url = form.api_url.data
                    config.api_key_encrypted = encrypted_key
                    config.model_name = form.model_name.data
                else:
                    config = AIConfig(
                        api_url=form.api_url.data,
                        api_key_encrypted=encrypted_key,
                        model_name=form.model_name.data
                    )
                    db.session.add(config)
                db.session.commit()
                flash("配置已保存", "success")
                return redirect(url_for("ai_config"))

        # Pre-populate form with existing config
        if config and not form.is_submitted():
            form.api_url.data = config.api_url
            form.model_name.data = config.model_name

        return render_template("config.html", ai_form=form, ai_config=config, form=form)
    ```

    Note: Test button validation happens first, save button second. Both require form validation.
