From exporters/base.py:
```python
class ExporterBase(ABC):
    def export(self, records: List[Any], **options) -> BytesIO:
        """Template method - calls _prepare_data() then _generate()"""

    @abstractmethod
    def _generate(self, data: List[Any], options: Dict) -> BytesIO:
        """Format-specific generation"""

    @property
    @abstractmethod
    def file_extension(self) -> str:
        """Return file extension without dot"""

    @property
    @abstractmethod
    def mime_type(self) -> str:
        """Return MIME type for send_file()"""
```
